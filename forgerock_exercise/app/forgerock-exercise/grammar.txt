value := word+ integer* | word* integer+ 
property := word+
word ::= [a-z]
integer ::= [0-9]
<exp> ::= BOOLEAN
<exp> ::= AND
<exp> ::= OR
<exp> ::= NOT
<exp> ::= PRESENT
<exp> ::= EQUALS
<exp> ::= LESS
<exp> ::= GREATER
<exp> ::= REGEX
<exp> ::= property <exp> value

OR ::= <exp> | <exp>
NOT ::= "!"<exp>

LESS ::= <exp> < <exp>
